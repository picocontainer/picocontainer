001    package org.picocontainer.web;

003    import org.picocontainer.behaviors.Storing;
004    import org.picocontainer.lifecycle.DefaultLifecycleState;
005    import org.picocontainer.lifecycle.LifecycleState;
006    
007    import java.io.Serializable;
008    
009    public class SessionStoreHolder implements Serializable {
010        private final Storing.StoreWrapper storeWrapper;
011        private final LifecycleState lifecycleState;
012    
013        public SessionStoreHolder(Storing.StoreWrapper storeWrapper,
014                                  LifecycleState lifecycleState) {
015            this.storeWrapper = storeWrapper;
016            this.lifecycleState = lifecycleState;
017        }
018    
019        Storing.StoreWrapper getStoreWrapper() {
020            return storeWrapper;
021        }
022    
023        LifecycleState getLifecycleState() {
024            return lifecycleState;
025        }
026    }
